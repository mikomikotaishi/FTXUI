24// Define a special style for some menu entry.
25MenuEntryOption Colored(ftxui::Color c) {
26 MenuEntryOption option;
27 option.animated_colors.foreground.enabled = true;
28 option.animated_colors.background.enabled = true;
29 option.animated_colors.background.active = c;
30 option.animated_colors.background.inactive = Color::Black;
31 option.animated_colors.foreground.active = Color::White;
32 option.animated_colors.foreground.inactive = c;
33 return option;
34}
