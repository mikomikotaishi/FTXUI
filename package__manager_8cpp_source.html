60 auto to_text = [](int number) {
61 return text(std::to_string(number)) | size(WIDTH, EQUAL, 3);

64 auto renderTask = [&](const Task& task) {
65 auto style = (task.downloaded == task.size) ? dim : bold;
66 return hbox({
67 text(task.name) | style,
68 separator(),
69 to_text(task.downloaded),
70 text("/"),
71 to_text(task.size),
72 separator(),
73 gauge(task.downloaded / float(task.size)),
74 });
75 };

77 auto renderSummary = [&]() {

96 auto render = [&]() {
97 std::vector<Element> entries;
98 for (auto& task : displayed_task)
99 entries.push_back(renderTask(task));
100
101 return vbox({
102 // List of tasks.
103 window(text(" Task "), vbox(std::move(entries))),

107 renderSummary(),
108 filler(),
109 }),

113 auto updateModel = [&]() {
114 for (auto& task : displayed_task) {
115 if (task.downloaded != task.size) {
116 task.downloaded++;
117 } else if (task.number_of_threads) {
118 remaining_threads += task.number_of_threads;
119 task.number_of_threads = 0;
120 nb_active--;
121 nb_done++;
122 }
123 }
124
125 if (remaining_tasks.size() &&
126 remaining_tasks.front().number_of_threads <= remaining_threads) {
127 remaining_threads -= remaining_tasks.front().number_of_threads;
128 displayed_task.push_back(remaining_tasks.front());
129 remaining_tasks.pop_front();
130 nb_queued--;
131 nb_active++;
132 }
133 };
134
135 std::string reset_position;
136 for (;;) {

145 // Simulate time.

148
149 // Exit
150 if (nb_active + nb_queued == 0)
151 break;
152
153 // Update the model for the next frame.
154 updateModel();
155 }
156 std::cout << std::endl;
157}
