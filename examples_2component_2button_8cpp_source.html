21// This is a helper function to create a button with a custom style.
22// The style is defined by a lambda function that takes an EntryState and
23// returns an Element.
24// We are using `center` to center the text inside the button, then `border` to
25// add a border around the button, and finally `flex` to make the button fill
26// the available space.
27ButtonOption ButtonStyle() {
28 auto option = ButtonOption::Animated();
29 option.transform = [](const EntryState& s) {
30 auto element = text(s.label);
31 if (s.focused) {
32 element |= bold;
33 }
34 return element | center | borderEmpty | flex;
35 };
36 return option;
37}

39int main() {
40 int value = 50;
41
42 // The tree of components. This defines how to navigate using the keyboard.
43 auto buttons = Container::Vertical({
44 Container::Horizontal({
45 Button(
46 "-1", [&] { value--; }, ButtonStyle()),
47 Button(
48 "+1", [&] { value++; }, ButtonStyle()),
49 }) | flex,
50 Container::Horizontal({
51 Button(
52 "-10", [&] { value -= 10; }, ButtonStyle()),
53 Button(
54 "+10", [&] { value += 10; }, ButtonStyle()),
55 }) | flex,
56 });
57
58 // Modify the way to render them on screen:
59 auto component = Renderer(buttons, [&] {
60 return vbox({
61 text("value = " + std::to_string(value)),
62 separator(),
63 buttons->Render() | flex,
64 }) |
65 flex | border;
66 });
67
68 auto screen = ScreenInteractive::Fullscreen();
69 screen.Loop(component);
70 return 0;
71}
