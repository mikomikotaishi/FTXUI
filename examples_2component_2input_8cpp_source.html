34 // The password input component:
35 InputOption password_option;
36 password_option.password = true;
37 Component input_password = Input(&password, "password", password_option);

39 // The phone number input component:
40 // We are using `CatchEvent` to filter out non-digit characters.
41 Component input_phone_number = Input(&phoneNumber, "phone number");
42 input_phone_number |= CatchEvent([&](Event event) {
43 return event.is_character() && !std::isdigit(event.character()[0]);
44 });
45 input_phone_number |= CatchEvent([&](Event event) {
46 return event.is_character() && phoneNumber.size() > 10;
47 });
48
49 // The component tree:
50 auto component = Container::Vertical({
51 input_first_name,
52 input_last_name,
53 input_password,
54 input_phone_number,
55 });
56
57 // Tweak how the component tree is rendered:
58 auto renderer = Renderer(component, [&] {
59 return vbox({
60 hbox(text(" First name : "), input_first_name->Render()),
61 hbox(text(" Last name : "), input_last_name->Render()),
62 hbox(text(" Password : "), input_password->Render()),
63 hbox(text(" Phone num : "), input_phone_number->Render()),
64 separator(),
65 text("Hello " + first_name + " " + last_name),
66 text("Your password is " + password),
67 text("Your phone number is " + phoneNumber),
68 }) |
69 border;
70 });
71
