23Element make_box(int x, int y) {
24 std::string title = "(" + std::to_string(x) + ", " + std::to_string(y) + ")";
25 return text(title) | center | size(WIDTH, EQUAL, 18) |
26 size(HEIGHT, EQUAL, 9) | border |
27 bgcolor(Color::HSV(x * 255 / 15, 255, y * 255 / 15));
28};

30Element make_grid() {
31 std::vector<Elements> rows;
32 for (int i = 0; i < 15; i++) {
33 std::vector<Element> cols;
34 for (int j = 0; j < 15; j++) {
35 cols.push_back(make_box(i, j));
36 }
37 rows.push_back(cols);
38 }
39
40 return gridbox(rows);
41};
