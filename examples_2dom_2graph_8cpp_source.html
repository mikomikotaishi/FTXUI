28class Graph {
29 public:
30 std::vector<int> operator()(int width, int height) const {
31 std::vector<int> output(width);
32 for (int i = 0; i < width; ++i) {
33 float v = 0;
34 v += 0.1f * sin((i + shift) * 0.1f); // NOLINT
35 v += 0.2f * sin((i + shift + 10) * 0.15f); // NOLINT
36 v += 0.1f * sin((i + shift) * 0.03f); // NOLINT
37 v *= height; // NOLINT
38 v += 0.5f * height; // NOLINT
39 output[i] = static_cast<int>(v);
40 }
41 return output;
42 }
43 int shift = 0;
44};

46std::vector<int> triangle(int width, int height) {
47 std::vector<int> output(width);
48 for (int i = 0; i < width; ++i) {
49 output[i] = i % (height - 4) + 2;
50 }
51 return output;
52}
